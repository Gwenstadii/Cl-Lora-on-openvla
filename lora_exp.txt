torchrun --standalone --nproc_per_node 1 vla-scripts/finetune.py \
 --vla_path "/root/autodl-tmp/models/openvla-7b" \
 --data_root_dir "/root/autodl-tmp/modified_libero_rlds" \
 --dataset_name "libero_spatial_no_noops" \
 --run_id_override "overfit_test_2000_steps_bs1" \
 --batch_size 1 \
 --grad_accumulation_steps 8 \
 --learning_rate 5e-4 \
 --use_lora True \
 --lora_rank 16 \

 --save_freq 2000 \
 --image_aug True \
 --run_root_dir "/root/autodl-tmp/LOGS"

  --pretrained_checkpoint /root/autodl-tmp/LOGS/overfit_test_2000_steps_bs1--2000_chkpt \
  --task_suite_name libero_spatial

运行
python -c "
import json

path_b = '/root/autodl-tmp/LOGS/task_b_cl_10k_from_90--2000_chkpt/dataset_statistics.json'

try:
    with open(path_b, 'r') as f: sb = json.load(f)
    for step in ['16000', '32000']:
        path_c = '/root/autodl-tmp/LOGS/normal_lora_taskC_object--' + step + '_chkpt/dataset_statistics.json'
        try:
            with open(path_c, 'r') as f: sc = json.load(f)
            sc.update(sb)
            with open(path_c, 'w') as f: json.dump(sc, f)
            print('Successfully merged dict for step ' + step)
        except Exception as e:
            print('Failed on step ' + step + ': ' + str(e))
except Exception as e:
    print('Error loading path_b: ' + str(e))
"
